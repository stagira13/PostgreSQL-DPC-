--退院日に出来高になってないもの

select データ識別番号,診療明細名称,使用量,明細点数・金額,行為回数,実施年月日,病棟コード
from etable
Where 実施年月日 = 退院年月日
AND 行為明細区分情報 LIKE '0_0_________'
AND データ区分 BETWEEN 21 AND 23
AND 明細点数・金額 > 0
AND 行為回数 > 1;

--退院日前日の退院処方探し（おそらく退院処方）

select データ識別番号,診療明細名称,使用量,明細点数・金額,行為回数,実施年月日,病棟コード
from etable
Where 退院年月日 - 実施年月日 = 1
AND 行為明細区分情報 LIKE '0_0_________'
AND データ区分 BETWEEN 21 AND 22
AND 明細点数・金額 > 0
AND 行為回数 > 1
AND 退院年月日 <> 'infinity'


--退院日以前の退院処方

select データ識別番号,診療明細名称,使用量,明細点数・金額,行為回数,実施年月日,病棟コード
from etable
Where 実施年月日 <> 退院年月日
AND 行為明細区分情報 LIKE '1_0_________'
AND データ区分 BETWEEN 21 AND 23
AND 明細点数・金額 > 0;
